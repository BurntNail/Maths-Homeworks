		|OA| &= \sqrt{1^2 + 2^2} = \sqrt{5} \\
		OA_\text{Unit} &= \frac{OA}{\sqrt{5}} \\
		&= \frac{1}{\sqrt{5}}i - \frac{2}{\sqrt{5}}j \\
		&= \frac{\sqrt{5}}{5}i - \frac{2\sqrt{5}}{5}j \\
		|AC| &= \frac{2\sqrt{5}}{5}i - \frac{4\sqrt{5}}{5}j \\ \\
		\text{Max} &= i - 2j + \frac{2\sqrt{5}}{5}i - \frac{4\sqrt{5}}{5}j \\
		\text{Min} &= i - 2j - \frac{2\sqrt{5}}{5}i + \frac{4\sqrt{5}}{5}j
	\end{align*}
}
