\qsn{4}{
\qsp{a}{
\begin{align*}
	F &= ma \\
	T - (110 + 190 + 1700)*9.8 &= (110 + 190 + 1700)*-1.8 \\
	T &= (110 + 190 + 1700) * (-9.8 - 1.8) \\
	T &= 23200N \\
\end{align*}
}
\qsp{b}{
	\begin{multicols}{2}
		\noindent
		\qsp{i}{
			\begin{align*}
				ma &= F \\
				110 * -1.8 &= F - 190*9.8 \\
				F &= - 190*9.8 + 110 * 1.8 \\
				F &= -1664 = 1660N \\
			\end{align*}
		}
		\columnbreak
		\qsp{ii}{
			\begin{align*}
				ma &= F \\
				1700 * -1.8 &= F - 110 * 9.8 - 190 * 9.8 \\
				F &= - 110 * 9.8 - 190 * 9.8 + 1700 * 1.8 \\
				F &= 120N \\
			\end{align*}
		}
	\end{multicols}
}
}
